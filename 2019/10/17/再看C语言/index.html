3 cin

3.1 简介（参考链接）

cin 是 C++ 的标准输入流对象，即 istream 类的对象。cin 主要用于从标准输入读取数据，这里的标准输入，指的是终端的键盘。此外，cout 是输出流对象，即 ostream 类的对象。这里的标准输出指的是终端屏幕。

在理解 cin 功能时，不得不提标准输入缓冲区。当我们从键盘输入字符串的时候$\color{red}{需要敲一下回车键才能够将这个字符串送入到缓冲区中}$，那么敲入的这个回车键（\r）会被转换为一个换行符（\n），这个换行符也会被存储在 cin 的缓冲区中并且被当成一个字符来计算！比如我们在键盘上敲下了 123456 这个字符串，然后敲一下回车键（\r）将这个字符串送入了缓冲区中，那么此时缓冲区中的字节个数是 7 ，而不是 6。

    }
}

4 sort

sort()定义在头文件中，接受两个或三个参数。当输入两个参数时，默认升序排序。若指定第三个参数则按照第三个参数指定的比较方式进行排序。

/*
*输入一个数组，排序后输出，处于中间位置的的数字。若一共有偶数个数字则输出中间两个数字。
*input: 1 3 5 7 2
*output:3
*/

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

bool compare(int a,int b) 
{ 
return a>b; //降序排列，如果改为return a<b，则为升序

} 

int main(){
    int temp;
    vector<int> arry;

    cout<<"please input a arry:"<<endl;
    while(cin>>temp){
        arry.push_back(temp);
        if(cin.get() == '\n')
            break;
    }
    cin.sync();

    cout<<"output"<<endl;
    cout<<"after sort:"<<endl;

    sort(arry.begin(), arry.end(), compare);


    for(int i = 0; i < arry.size(); i++)
        cout<<arry[i]<<" ";
    cout<<endl;

    if(arry.size()%2 == 0)
        cout<<arry[arry.size()/2 -1]<<" "<<arry[arry.size()/2 + 1 -1]<<endl;
    else
        cout<<arry[(int)arry.size()/2]<<endl;

    return 0;
}

注意到代码中compare函数中“a>b”，对应的时降序。此处，不加考证，直观的假设为，sort函数的排序是基于小于运算进行升序排序的。当指定了第三个参数时，再排序时对小于运算符进行了重载。如果将小于运算重载为大于运算后，再按照此时的“小于运算”进行排序的得到的就是降序。

此外sort排序可以用于更加复杂的排序，如自定义的结构体的排序。

bool dfs(int x, int y, int direct[], int count){//遍历各个方向
    count++;
    if(count == 5){
        cout<<vec[x][y]<<":("<<x - 4*direct[0]<<","<<y - 4*direct[1]<<")"<<endl;
        return true;
    }

    if(vec[x][y] == 0)
        return false;
    else if(vec[x][y] == 1){
        if(vec[x + direct[0]][y + direct[1]] == vec[x][y]){
            if(dfs(x + direct[0], y + direct[1], direct, count))
                return true;            
        }
    }
    else if(vec[x][y] == 2){
        if(vec[x + direct[0]][y + direct[1]] == vec[x][y]){
            if(dfs(x + direct[0], y + direct[1], direct, count))
                return true;        
        }    
    }
    return false;
}

输出：在标准输出上输出最后一个皇后放置的列数，若无解，输出字符串“No Answer”
样例：

输入        输出
627*4853    1
3681*752    4
1357246*    No Answer
*/
#include <iostream>
#include <vector>
#include <string>

using namespace std;

#define row 8
#define column 8

void zhanling(string str, vector<vector<int> >& vec){

    for(int i = 0; i < row; i++){
        if(str[i] != '*'){
            int lie = str[i] - '1';
            for(int j = 0; j < column; j++){
                vec[i][j] = 1;//第i行置1
                vec[j][lie] = 1;//第i列置1

                for(int k1 = 0; k1 < row; k1++){//重新写一个循环是因为,在之前的循环中是在固定的行中,但是此处需要遍历整个数组,所以重新写一个独立的循环
                    for(int k2 = 0; k2 < column; k2++){
                        if(k1 - k2 == i - lie)//（i，j）的正对角线置1
                            vec[k1][k2] = 1;
                        if(k1 + k2 == i + lie)//（i，j）的反对角线置1
                            vec[k1][k2] = 1;    
                    }
                }
            }        
        }
    }
}

int main(){

    string str;
    bool flag = false;
    vector<vector<int> > vec;

    for(int i = 0; i < row; i++){
        vector<int> tempVec;
        for(int j = 0; j < column; j++)
            tempVec.push_back(0);
        vec.push_back(tempVec);
    }

    cout<<"请输入（已正确放置的皇后的列数，未放置皇后的行用字符“*”来表示）："<<endl;
    cin>>str;

    zhanling(str, vec);

    for(int i1 = 0; i1 < row; i1++){
        for(int j1 = 0; j1 < column; j1++){
            //cout<<vec[i1][j1]<<" ";        
            if(vec[i1][j1] != 1){
                flag = true;
                cout<<j1 + 1<<endl;            
            }
        }
    }

    if(!flag)
        cout<<"No Ansower"<<endl;


    return 0;
}

/*
输入一篇可能未经排版的文章，挑选出其中的单词
【单词中不包含“（”等特殊符号】，
然后按字典序输出 单词及其频次。
如: 
输入:
i knowe you, do you knowe me?
输出:
do = 1
i = 1
konwe = 2
me = 1
you = 2
*/

#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;

void inputPerprocessing(vector<string>& vec){
    char temp;
    vec.push_back("\0");
    while(cin.get(temp)){
        if(isalpha(temp))
            vec[vec.size() -1] += temp;
        else
            vec.push_back("\0");
    }
    vec.erase(remove(vec.begin(), vec.end(), "\0"), vec.end());
}

bool compare(string a, string b){

    for(int i = 0; i < a.length() && i < b.length(); i++){
        tolower(a[i]);//因为不准备区分大小写,所以先把字符转为小写再对比
        tolower(b[i]);

        if(a[i] == b[i])//如果当前字符相等对比下一个字符
            continue;
        else
            return a[i]<b[i];
    }
    return a[0] < b[0];
}


int main(){
    vector<string> vec;//存储字符串中各个单词

    inputPerprocessing(vec);
    //cout<<vec.size()<<endl;


    sort(vec.begin(), vec.end(), compare);//按照字典序排序。

    for(int i = 0; i < vec.size(); i++){
        int temp;
        temp = count(vec.begin(), vec.end(), vec[i]);//统计当前单词的频次
        if(temp > 1){
            cout<<vec[i]<<"="<<temp<<endl;
            vec.erase(remove(vec.begin(), vec.end(), vec[i]), vec.end());//移出已经统计过的单词
            i--;//由于当前单词也被删除，所以i退一步
        }
        else
            cout<<vec[i]<<"="<<temp<<endl;//如果当前单词只出现了一次，则不删除也不会影响输出。

    }

    return 0;
}

#include <iostream>
#include <string>
#include <vector>


using namespace std;

void inputPreprocessing(vector<string>& vec){
    vec.push_back("\0");
    char temp;
    bool begin = false, end = false;
    cout<<"请输入enum语句:"<<endl;

    while(1){
        cin.get(temp);
        if(temp == '{')//当读到'{'时,开始存储需要的数据
            begin = true;
        while(begin){//开始存储
            cin.get(temp);
            if(temp == '}'){//当读到'{'时, 需要的数据已经读完,存储结束
                end = true;//在此处标记用于退出外层循环( "while(1)" )
                break;    //存储结束,退出循环
            }
            else if(temp == ',')
                vec.push_back("\0");
            else
                vec[vec.size() - 1] += temp;
        }

        if(end)
            break;
    }
}

int main(){

    int num = 0;

    vector<string> vec;//输入预处理

    inputPreprocessing(vec);

    for(int i = 0; i < vec.size(); i++){

        int temp = vec[i].find('=', 0);

        if(temp != -1){//如果有"="号
            num = 0;//初始化num,便于根据等号后边的数字重新赋值
            for(int k = temp +1; k < vec[i].size(); k++)//计算等号后字符对应的整数,
                num = num*10 + (vec[i][k] - '0');
            for(int j = 0; j < temp; j++)//带有等号时,输出需要特殊处理(=号及之后的字符不用输出)
                cout<<vec[i][j];
            cout<<"="<<num<<endl;
            num++;//控制num自增
            continue;
        }        
        cout<<vec[i]<<":"<<num<<endl;//如果没有等号

        num++;
    }

    return 0;
}

vector<char> fanzhuan(vector<char> vec){//求vec的逆序数
    vector<char> temp;
    for(int i = 0; i < vec.size(); i++)
        temp.push_back(vec[vec.size() - 1 - i]);

    return temp;
}

void nixu(vector<vector<char> > vec, vector<vector<char> >& vec2){

    for(int i = 0; i < vec.size(); i++)//遍历寻找逆序数
        vec2.push_back(fanzhuan(vec[i]));
}
